function getResumenJugadoresCompleto() { const sheet = getJugadoresSheet(); const data = sheet.getDataRange().getValues(); const jugadores = []; for (let i = 1; i < data.length; i++) { jugadores.push({ cedula: data[i][4], primerNombre: data[i][1] || '', segundoNombre: data[i][2] || '', apellidos: data[i][3] || '', nombreCamiseta: data[i][16] || '', dorsal: data[i][17] || '', talla: data[i][27] || '', inscripEstado: data[i][23] || 'PENDIENTE', unifEstado: data[i][24] || 'PENDIENTE', mediasEstado: data[i][25] || 'PENDIENTE' }); } return json({ ok: true, jugadores }); }
