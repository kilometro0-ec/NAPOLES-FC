const SPREADSHEET_ID = "1RIW76k-tLe41P6mf0PPUStKcchMYLt7sHTOaN0MwFZM"; const ROOT_FOLDER_ID = "11rYr6psxvsYmFn3H57Vna7qdeNSsWoe9"; // Carpeta "JUGADORES NAPOLES F.C" // ================= GET ================= function doGet(e) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheetJugadores = ss.getSheetByName("Jugadores"); const action = e?.parameter?.action; // Servir el HTML del panel de administración if (action === "admin") { return HtmlService.createHtmlOutputFromFile("admin") .setTitle("Panel Administración") .addMetaTag("viewport", "width=device-width, initial-scale=1") .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // Validar si cédula ya existe if (action === "validarRegistro") { const cedula = e?.parameter?.cedula; if (!cedula) return json({ error: "Cédula requerida" }); const data = sheetJugadores.getDataRange().getValues(); let cedulaExiste = false; for (let i = 1; i < data.length; i++) { const cedGuardada = String(data[i][4]).replace(/^'/, '').trim(); if (cedGuardada === cedula) { cedulaExiste = true; break; } } return json({ cedulaExiste }); } // Obtener dorsales ocupados if (action === "getDorsales") { if (!sheetJugadores) { return json([]); } const data = sheetJugadores.getDataRange().getValues(); const ocupados = []; for (let i = 1; i < data.length; i++) { const dorsal = data[i][17]; if (dorsal && dorsal.toString().trim()) { ocupados.push(dorsal.toString().trim()); } } return json(ocupados); } return json({ ok: true, msg: "API funcionando" }); } // ================= POST ================= function doPost(e) { const lock = LockService.getScriptLock(); try { lock.waitLock(15000); // Parsear datos let params = {}; if (e && e.parameter) { params = e.parameter; } const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName("Jugadores"); // Crear hoja si no existe if (!sheet) { sheet = ss.insertSheet("Jugadores"); sheet.appendRow([ "Timestamp", "Primer Nombre", "Segundo Nombre", "Apellidos", "Cédula", "Fecha Nacimiento", "Teléfono", "Correo", "Foto Perfil", "Foto Cedula", "Estado", "Goles", "Amarillas", "Rojas", "Deuda", "Inscripción", "Uniforme", "Dorsal", "Nombre Camiseta", "Medias Extras", "Fecha Registro" ]); } // Crear nombre de carpeta para el jugador const nombreCompleto = `${params.n1 || ""} ${params.n2 || ""} ${params.ape || ""}`.trim().replace(/\s+/g, " "); const nombreCarpeta = `${nombreCompleto} - ${params.ced}`; // Procesar imágenes (guardar en Drive dentro de la subcarpeta del jugador) let fotoUrl = ""; let fotoCedulaUrl = ""; if (params.fotoRostroB64) { fotoUrl = saveImageToPlayerFolder(params.fotoRostroB64, nombreCarpeta, params.ced, "perfil"); } if (params.fotoCedulaB64) { fotoCedulaUrl = saveImageToPlayerFolder(params.fotoCedulaB64, nombreCarpeta, params.ced, "cedula"); } // Agregar jugador a la hoja de cálculo sheet.appendRow([ new Date(), // Timestamp params.n1 || "", // Primer Nombre params.n2 || "", // Segundo Nombre params.ape || "", // Apellidos params.ced || "", // Cédula params.fecha || "", // Fecha Nacimiento params.tel || "", // Teléfono params.correo || "", // Correo fotoUrl, // Foto Perfil fotoCedulaUrl, // Foto Cedula "PENDIENTE", // Estado 0, // Goles 0, // Amarillas 0, // Rojas 0, // Deuda params.inscripcion || "", // Inscripción params.uniforme || "", // Uniforme params.dorsal || "", // Dorsal params.nombreCamiseta || "", // Nombre Camiseta params.medias || "NO", // Medias Extras new Date() // Fecha Registro ]); return json({ ok: true, msg: "Jugador registrado exitosamente" }); } catch (err) { Logger.log("ERROR: " + err); return json({ ok: false, error: err.toString() }); } finally { lock.releaseLock(); } } // ================= FUNCIÓN PARA GUARDAR IMÁGENES EN CARPETA DEL JUGADOR ================= function saveImageToPlayerFolder(base64Data, nombreCarpeta, cedula, tipo) { try { // Extraer el tipo de imagen y los datos const matches = base64Data.match(/^data:image\/(\w+);base64,(.+)$/); if (!matches) return ""; const extension = matches[1]; const imageData = matches[2]; // Obtener la carpeta raíz "JUGADORES NAPOLES F.C" let rootFolder; try { rootFolder = DriveApp.getFolderById(ROOT_FOLDER_ID); Logger.log("Carpeta raíz encontrada: " + rootFolder.getName()); } catch (e) { Logger.log("Error al obtener carpeta raíz: " + e); // Si no existe, crearla rootFolder = DriveApp.createFolder("JUGADORES NAPOLES F.C"); Logger.log("Carpeta raíz creada: " + rootFolder.getName()); } // Buscar si ya existe una carpeta con el nombre del jugador let playerFolder = null; const folders = rootFolder.getFolders(); while (folders.hasNext()) { const folder = folders.next(); if (folder.getName() === nombreCarpeta) { playerFolder = folder; Logger.log("Carpeta existente encontrada: " + nombreCarpeta); break; } } // Si no existe, crear la carpeta del jugador if (!playerFolder) { playerFolder = rootFolder.createFolder(nombreCarpeta); Logger.log("Nueva carpeta creada: " + nombreCarpeta); } // Crear nombre de archivo const nombreArchivo = `${cedula}_${tipo}_${new Date().getTime()}.${extension}`; // Decodificar base64 y crear blob const blob = Utilities.newBlob(Utilities.base64Decode(imageData), `image/${extension}`, nombreArchivo); // Guardar archivo en la carpeta del jugador const file = playerFolder.createFile(blob); file.setDescription(`Foto ${tipo} del jugador ${cedula}`); Logger.log("Archivo guardado: " + nombreArchivo); // Retornar URL de vista return `https://drive.google.com/uc?export=view&id=${file.getId()}`; } catch (err) { Logger.log("Error guardando imagen: " + err); return ""; } } // ================= UTIL ================= function json(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ================= FUNCIÓN PARA VERIFICAR CARPETA (OPCIONAL) ================= function verificarCarpetaRaiz() { try { let folder; try { folder = DriveApp.getFolderById(ROOT_FOLDER_ID); Logger.log("Carpeta encontrada: " + folder.getName() + " - ID: " + folder.getId()); } catch (e) { folder = DriveApp.createFolder("JUGADORES NAPOLES F.C"); Logger.log("Carpeta creada: " + folder.getName() + " - ID: " + folder.getId()); } return folder.getId(); } catch (err) { Logger.log("Error: " + err); return null; } }
